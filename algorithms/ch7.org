(require 'cl)
(defun upper-bound (m)
  (let ((n 100000000))
    (floor (log (/ (1+ n) 4.0) (ceiling (/ m 2.0))))))

(upper-bound 1000)
2

(loop for m from 1000 downto 1
      when (> (upper-bound m) 2)
      return m)
584
